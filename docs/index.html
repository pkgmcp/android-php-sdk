Linux udev Setup (fastboot-php)

⚠️
On Linux, you need udev rules to access USB devices without root. Create /etc/udev/rules.d/51-android.rules.
udev
# Google / Pixel
SUBSYSTEM=="usb", ATTR{idVendor}=="18d1", MODE="0666", GROUP="plugdev"
# Samsung
SUBSYSTEM=="usb", ATTR{idVendor}=="04e8", MODE="0666", GROUP="plugdev"
# OnePlus
SUBSYSTEM=="usb", ATTR{idVendor}=="2a70", MODE="0666", GROUP="plugdev"
bash
sudo udevadm control --reload-rules && sudo udevadm trigger
sudo usermod -aG plugdev $USER

Quick Start

fastboot-php — Flash a Boot Image

php
use FastbootPhp\FastbootDevice;
use FastbootPhp\Common;
use FastbootPhp\Transport\LibUsbTransport;

Common::setDebugLevel(Common::LEVEL_DEBUG);

$device = new FastbootDevice(
    new LibUsbTransport('/dev/bus/usb/001/005')
);

$device->connect();

// Read device info
$product = $device->getVariable('product');
$slot    = $device->getVariable('current-slot');
echo "Device: {$product} (slot {$slot})\n";

// Flash boot partition with progress
$device->flashBlob(
    partition:  'boot',
    data:       file_get_contents('boot.img'),
    onProgress: fn(float $p) => printf("%.0f%%\r", $p * 100),
);

$device->reboot();
$device->disconnect();

adb-php — Shell, Install, Screenshot

php
use AdbPhp\AdbClient;

$adb     = AdbClient::create();
$devices = $adb->listDevices();

$device = $adb->getDevice($devices[0]->id);

// Shell command
echo $device->shell('uname -a');

// Install APK
$device->install('/path/to/app.apk');

// Screenshot
file_put_contents('/tmp/screen.png', $device->screencap());

// Push file
$device->push('/local/data.db', '/sdcard/data.db', progress: fn($b) => printf("%d bytes\r", $b));

// Stream logcat
$reader = $device->openLogcat();
$reader->stream(function($entry): bool {
    echo $entry; // calls __toString()
    return true;
});
$reader->end();

Wireless ADB

php
$adb = AdbClient::create();
echo $adb->connect('192.168.1.100', 5555);

$device = $adb->getDevice('192.168.1.100:5555');
echo $device->shell('id');

SyncService

Low-level SYNC protocol for direct file operations. Obtain via $device->syncService(). Always call end() when done.

Typed Constants

php
SyncService::DEFAULT_MODE; // int: 0o644

Example

php
$sync = $device->syncService();

// Stat
$entry = $sync->stat('/sdcard/file.txt');
echo $entry->size;

// Readdir
$entries = $sync->readdir('/sdcard');
foreach ($entries as $e) {
    echo ($e->isDirectory() ? 'd' : '-') . ' ' . $e->name . "\n";
}

// Push file
$sync->pushFile('/local/app.db', '/sdcard/app.db');

// Pull file
['data' => $bytes] = $sync->pull('/sdcard/app.db');

$sync->end(); // Always close!

Logcat

Read and parse the Android binary logcat stream.

Priority Constants (typed int)

php
LogcatEntry::PRIORITY_VERBOSE // 2 → 'V'
LogcatEntry::PRIORITY_DEBUG   // 3 → 'D'
LogcatEntry::PRIORITY_INFO    // 4 → 'I'
LogcatEntry::PRIORITY_WARN    // 5 → 'W'
LogcatEntry::PRIORITY_ERROR   // 6 → 'E'
LogcatEntry::PRIORITY_FATAL   // 7 → 'F'
LogcatEntry::PRIORITY_SILENT  // 8 → 'S'

Streaming Example

php
$reader = $device->openLogcat(['clear' => true]);

$reader->stream(function(LogcatEntry $e): bool {
    if ($e->priority >= LogcatEntry::PRIORITY_WARN) {
        printf("[%s/%s] %s\n", $e->priorityLabel(), $e->tag, $e->message);
    }
    return true; // false stops streaming
});

$reader->end();

Error Handling

fastboot-php

php
use FastbootPhp\{FastbootError, UsbError};

try {
    $device->connect();
    $device->flashBlob('boot', file_get_contents('boot.img'));
    $device->reboot();
} catch (UsbError $e) {
    // Hardware / transport problem
    fwrite(STDERR, "USB: " . $e->getMessage());
} catch (FastbootError $e) {
    // Bootloader rejected the command
    fwrite(STDERR, "Bootloader [{$e->status}]: {$e->bootloaderMessage}");
} finally {
    $device->disconnect();
}

adb-php

php
use AdbPhp\Exceptions\{AdbException, ConnectionException};

try {
    $device->install('/path/app.apk');
} catch (ConnectionException $e) {
    echo "ADB server not running: " . $e->getMessage();
} catch (AdbException $e) {
    echo "ADB error: " . $e->getMessage();
}
